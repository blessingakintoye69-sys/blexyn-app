const express = require("express"); const http = require("http"); const socketIo = require("socket.io"); const multer = require("multer"); const path = require("path"); const app = express(); const server = http.createServer(app); const io = socketIo(server); app.use(express.json()); app.use("/uploads", express.static(path.join(__dirname, "uploads"))); let users = []; let messages = []; let currentId = 1; // Storage for images const storage = multer.diskStorage({ destination: "./uploads", filename: (req, file, cb) => { cb(null, Date.now() + ".jpg"); } }); const upload = multer({ storage }); // FRONTEND (ALL PAGES INSIDE ONE) app.get("/", (req, res) => { res.send(` Dating App

Dating App ❤️

Register





Login

Users

Chat

`); }); // Register app.post("/register", upload.single("image"), (req, res) => { const user = { id: currentId++, name: req.body.name, age: req.body.age, bio: req.body.bio, image: req.file ? req.file.filename : "", likes: [], matches: [] }; users.push(user); res.send(user); }); // Login app.post("/login", (req, res) => { const user = users.find(u => u.name === req.body.name); res.send(user || { error: "User not found" }); }); // Users app.get("/users", (req, res) => { res.send(users); }); // Like + Match app.post("/like", (req, res) => { const { userId, likedUserId } = req.body; const user = users.find(u => u.id == userId); const liked = users.find(u => u.id == likedUserId); if (!user.likes.includes(likedUserId)) { user.likes.push(likedUserId); } if (liked.likes.includes(userId)) { if (!user.matches.includes(likedUserId)) { user.matches.push(likedUserId); liked.matches.push(userId); } } res.send({ success: true }); }); // Chat (Socket) io.on("connection", (socket) => { socket.on("sendMessage", (data) => { messages.push(data); io.emit("receiveMessage", data); }); }); server.listen(3000, () => { console.log("Running on http://localhost:3000"); });
