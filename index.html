// ================= BACKEND (Node.js + Express) ================= const express = require('express'); const app = express(); app.use(express.json()); // Fake database let users = []; let transactions = []; // Sample Data Plans const DATA_PLANS = { MTN: [ { plan: "500MB", price: 180 }, { plan: "1GB", price: 300 }, { plan: "2GB", price: 600 } ], AIRTEL: [ { plan: "1GB", price: 350 }, { plan: "2GB", price: 700 } ] }; // Register app.post('/register', (req, res) => { const { phone, password } = req.body; users.push({ phone, password, balance: 0 }); res.json({ message: 'Registered successfully' }); }); // Login app.post('/login', (req, res) => { const { phone, password } = req.body; const user = users.find(u => u.phone === phone && u.password === password); if (!user) return res.status(400).json({ error: 'Invalid login' }); res.json({ message: 'Login successful', user }); }); // Get Data Plans app.get('/plans/:network', (req, res) => { const network = req.params.network.toUpperCase(); res.json(DATA_PLANS[network] || []); }); // Fund Wallet app.post('/fund', (req, res) => { const { phone, amount } = req.body; const user = users.find(u => u.phone === phone); user.balance += amount; res.json({ balance: user.balance }); }); // Buy Data app.post('/buy-data', (req, res) => { const { phone, network, plan } = req.body; const user = users.find(u => u.phone === phone); const selectedPlan = DATA_PLANS[network].find(p => p.plan === plan); if (user.balance < selectedPlan.price) { return res.status(400).json({ error: 'Insufficient balance' }); } user.balance -= selectedPlan.price; transactions.push({ phone, network, plan, price: selectedPlan.price }); res.json({ message: 'Data purchased', balance: user.balance }); }); // Transactions app.get('/transactions/:phone', (req, res) => { const userTx = transactions.filter(t => t.phone === req.params.phone); res.json(userTx); }); app.listen(3000, () => console.log('Server running on port 3000')); // ================= FRONTEND (HTML + JS) ================= /* Data App
